   - Technical requirements

5. Saves results to 'job_analysis.json'

#!/usr/bin/env python3
"""
Automated Job Analysis Tool
Analyzes Upwork jobs and recommends optimal bids
"""

import json
import re
from pathlib import Path

class JobAnalyzer:
    def __init__(self):
        self.urgency_keywords = [
            'urgent', 'asap', 'immediate', 'quickly', 'fast',
            'today', 'now', 'rush', 'emergency', 'critical'
        ]
    
    def extract_budget(self, text):
        """Extract budget from job description"""
        # Matches $100, $1,000, $100-500, etc.
        pattern = r'\$(\d{1,3}(?:,\d{3})*(?:\.\d{2})?)'
        matches = re.findall(pattern, text)
        if matches:
            # Convert to integers
            budgets = [int(m.replace(',', '')) for m in matches]
            return {'min': min(budgets), 'max': max(budgets)}
        return None
    
    def calculate_urgency(self, text):
        """Calculate urgency score 0-100"""
        text_lower = text.lower()
        urgency_count = sum(1 for kw in self.urgency_keywords if kw in text_lower)
        return min(urgency_count * 20, 100)
    
    def calculate_win_probability(self, budget, urgency):
        """Calculate win probability based on budget and urgency"""
        score = 50  # Base score
        
        # Budget scoring (your sweet spot: $200-$500)
        if budget:
            if 200 <= budget['min'] <= 500:
                score += 30
            elif 150 <= budget['min'] <= 200:
                score += 20
            elif budget['min'] < 150:
                score += 5  # Too low
            else:
                score += 10  # High budget
        
        # Urgency scoring
        score += (urgency * 0.2)  # Max +20 points
        
        return min(score, 100)
    
    def recommend_bid(self, budget):
        """Recommend bid amount"""
        if not budget:
            return 300  # Default
        
        # Bid slightly under budget to be competitive
        mid_point = (budget['min'] + budget['max']) / 2
        recommended = int(mid_point * 0.95)  # 5% discount
        
        # Round to nearest $10
        recommended = (recommended // 10) * 10
        
        return max(200, min(recommended, 500))  # Keep in range
    
    def analyze_job(self, description):
        """Analyze a single job description"""
        budget = self.extract_budget(description)
        urgency = self.calculate_urgency(description)
        win_prob = self.calculate_win_probability(
            budget, urgency
        )
        bid = self.recommend_bid(budget)
        
        return {
            'description': description[:200] + '...',
            'budget': budget,
            'urgency_score': urgency,
            'win_probability': win_prob,
            'recommended_bid': bid,
            'should_apply': win_prob >= 60
        }
    
    def process_file(self, input_file, output_file):
        """Process job file and save results"""
        try:
            # Read jobs
            with open(input_file, 'r', encoding='utf-8') as f:
                jobs = [line.strip() for line in f if line.strip()]
            
            # Analyze each
            results = []
            for i, job in enumerate(jobs, 1):
                analysis = self.analyze_job(job)
                analysis['job_number'] = i
                results.append(analysis)
                
                print(f"Job {i}: Win Prob {analysis['win_probability']}% "
                      f"â†’ Bid ${analysis['recommended_bid']}")
            
            # Save results
            with open(output_file, 'w', encoding='utf-8') as f:
                json.dump(results, f, indent=2)
            
            print(f"\nâœ… Analyzed {len(results)} jobs")
            print(f"âœ… Saved results to {output_file}")
            
            # Summary
            high_prob = [r for r in results if r['win_probability'] >= 70]
            print(f"\nğŸ¯ {len(high_prob)} HIGH-PROBABILITY jobs (>70%)")
            
            return results
            
        except FileNotFoundError:
            print(f"âŒ Error: {input_file} not found")
            return []
        except Exception as e:
            print(f"âŒ Error: {e}")
            return []

if __name__ == '__main__':
    analyzer = JobAnalyzer()
    analyzer.process_file('upwork_jobs.txt', 'job_analysis.json')
