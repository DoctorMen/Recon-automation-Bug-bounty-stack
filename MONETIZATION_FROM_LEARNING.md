```python
# monetization_analyzer.py

import json
from datetime import datetime

def analyze_learning_for_money(learning_data):
    """
    Analyze learning patterns and suggest monetization
    """
    
    # Calculate total compound effect
    total_compound = sum(entry['compoundEffect'] for entry in learning_data)
    
    # Find high-impact areas
    high_impact = [e for e in learning_data if e['impact'] == 'high']
    
    # Cluster by category
    categories = {}
    for entry in learning_data:
        cat = entry.get('category', 'General')
        categories[cat] = categories.get(cat, []) + [entry]
    
    # Generate strategies
    strategies = []
    
    # Strategy 1: If compound > 30x â†’ Premium consulting
    if total_compound > 30:
        strategies.append({
            'type': 'Premium Consulting',
            'reasoning': f'{total_compound}x compound = expert level',
            'pricing': '$200-500/hour',
            'potential': '$100,000-250,000/year'
        })
    
    # Strategy 2: If 5+ high impact â†’ Course creation
    if len(high_impact) >= 5:
        strategies.append({
            'type': 'Online Course',
            'reasoning': f'{len(high_impact)} high-impact skills to teach',
            'pricing': '$497-997',
            'potential': '$50,000-100,000 (100 sales)'
        })
    
    # Strategy 3: If 3+ categories â†’ Multiple revenue streams
    if len(categories) >= 3:
        strategies.append({
            'type': 'Multiple Services',
            'reasoning': f'{len(categories)} distinct expertise areas',
            'pricing': '$150-300/hour per stream',
            'potential': f'${len(categories) * 5000}/month'
        })
    
    # Strategy 4: Pattern-based products
    # If learned automation â†’ Sell automation tools
    automation_count = sum(1 for e in learning_data if 'automation' in e['title'].lower())
    if automation_count >= 3:
        strategies.append({
            'type': 'Automation Tools/SaaS',
            'reasoning': f'{automation_count} automation skills = productizable',
            'pricing': '$297-997/month subscription',
            'potential': '$30,000-100,000/month (100 customers)'
        })
    
    return {
        'total_compound': total_compound,
        'high_impact_count': len(high_impact),
        'categories': len(categories),
        'strategies': strategies
    }

# Usage
with open('output/learning_data.json') as f:
    data = json.load(f)

results = analyze_learning_for_money(data['entries'])
print(json.dumps(results, indent=2))
```
